# 下载xls文件

axios，设置:responseType: arraybuffer

列如:
```javascript
// 将返回数据类型设置成流数据类型，用于接收数据
export const distributorRegister = (params)=>{
  return fetch({
    url: '/distributor/get/register',
    params,
    method: 'get',
    responseType: 'arraybuffer'
  })
}
```


```javascript
// 下载函数
export const downLoadFile = (fileData,name)=>{
    let blob     = new Blob([fileData],{'type':'application/msexcel'});  // [文件]{类型}
    // 创建a标签模拟点击事件并下载
    let a  = document.createElement('a');
    a.download = `${name}.xls`  // 下载的文件名
    a.href     = window.URL.createObjectURL(blob);
    var evt = document.createEvent("MouseEvents");
    evt.initEvent("click", false, false);
    a.dispatchEvent(evt);
}
```
